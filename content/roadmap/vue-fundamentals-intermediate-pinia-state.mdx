export const useUserStore = defineStore('user', {
  state: () => ({ user: null, token: null }),
  getters: {
    isLoggedIn: (s) => !!s.token,
  },
  actions: {
    async login(email, password) {
      const res = await api.post('/auth/login', { email, password })
      this.user = res.user
      this.token = res.token
    },
  },
})
