## Vue Composition API

<LearningObjectives objectives={["Use ref and reactive correctly", "Derive state with computed", "React to changes with watch/watchEffect", "Extract composables for reuse"]} />

```vue
<script setup>
import { ref, computed, watch } from 'vue'

const count = ref(0)
const doubled = computed(() => count.value * 2)

watch(count, (n) => console.log('count:', n))

function increment() { count.value++ }
</script>
```

```js
// composables/useFetch.js
export function useFetch(url) {
  const data = ref(null)
  const error = ref(null)
  watchEffect(async () => {
    try { data.value = await (await fetch(url)).json() }
    catch (e) { error.value = e }
  })
  return { data, error }
}
```
